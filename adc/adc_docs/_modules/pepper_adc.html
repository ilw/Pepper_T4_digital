class AdcBuffer:
    def __init__(self, vdd = 1.8, t_settle = 0.5/(144e3), t_tau = 5, c_load = 2*15e-12):
        self.vdd = Quantity(vdd,units="V")
        self.t_settle = Quantity(t_settle,units="s")
        self.t_tau = Quantity(t_tau,units="")
        self.c_load = Quantity(c_load,units="F")

    @property
    def _time_constant(self):
        return self.t_settle.scale(1/self.t_tau)

    @property
    def bandwidth_ugf(self):
        return Quantity(1/(2*pi*self._time_constant),units="Hz")

    @property
    def _gm(self):
        return Quantity(self.c_load / self._time_constant,units="S")

    @property
    def _ibias(self):
        return Quantity(self._gm / 10, units="A")

    @property
    def power(self):
        return Quantity(self.vdd * self._ibias * 1.5, units="W")

def high_precision_normal(mu=mpf(0), sigma=mpf(1)):
    # Use Python's random module for uniform [0, 1) sampling
    # but cast to mpmath mpf for high-precision computation
    u1 = rand()
    u2 = rand()

    # Box-Muller transform
    z0 = sqrt(-2 * ln(u1)) * cos(2 * pi * u2)
    return mu + sigma * z0

    def convert(self, signal: np.array):
        assert signal.size == self.osr
        self.adc.zero_qnf()
        self.cic.zero_data()
        for index in range(self.osr):
            result = self.conversion(signal[index])
            acc_result = self.cic.update(result)
            if index == self.osr // 2:
                mid_point = acc_result
        return (2 * mid_point - acc_result) / self.osr ** 2



    @property
    def switching_energy(self) -> float:
        """Switching energy dissipated per sample."""
        return self.adc.switching_energy * self.osr

    @property
    def analog_energy(self) -> float:
        """Analog energy dissipated per sample."""
        return self.adc.analog_energy * self.osr
    
    @property
    def digital_energy(self) -> float:
        """Digital energy dissipated per sample."""
        return self.adc.digital_energy * self.osr

    @property
    def total_energy(self) -> float:
        """Total energy dissipated per sample."""
        return (
            self.switching_energy
            + self.analog_energy
            + self.digital_energy
        )

class PepperDataConverter_V1:

    """
    Models the Pepper capacitor CDAC array with MES2+DWA functionality.

    This is a single-ended representation of the fully-differential 
    capacitive dac used in the Pepper NS ADC. Metrics will see 6dB improvement.

    The design does not use split capacitors and instead segments the 
    array into 4 sections: one DWA MSBs, two MES2 LSBs, and coupling to
    the analogue quantization noise-shaping filter.

    The MSB section uses two 832 fF unit capacitors for feeding back the 
    first 4 bits. The same 832 fF capacitor is used to couple the analogue
    loopfilter during sampling. This is done so that the common-mode reference
    voltage is used as little as possible and kept quiet.

    The LSB section uses two 6 bit CDACs with a ping-pong configuration.
    Here the unit capacitor is 51.4 fF. 

    The total sampling capacitance is 15x 2x 832 fF or around 25 pF on both
    positive and negative analogue inputs.
    """

    _k = mpf(constants.k)  # Boltzmann constant
    _T = mpf(constants.convert_temperature(25.0, "Celsius", "Kelvin"))  # Convert to Kelvin
    _kT = _k * _T
    _amplifier_gain = 10 ** (50/20) # 50 dB target

    def __init__(self, 
            vdd: float = 1.8,
        ):
        self.N = 10
        self.msb_unit_capacitor = mpf(2*832.0e-15)
        self.lsb_unit_capacitor = mpf(0.5*52.0e-15)
        self._v_sampled = 0.0
        self._c_density = mpf(2e-15)
        self._vdd = mpf(vdd)
        self._lsb_bits = 6
        self._msb_bits = 4
        self._msb_dout = np.repeat(mpf(0.0), self._msb_bits)
        self._lsb_sd1_dout = np.repeat(mpf(0.0), self._lsb_bits)
        self._lsb_sd2_dout = np.repeat(mpf(0.0), self._lsb_bits)
        self._qnf_aout = np.repeat(mpf(0.0), 2)
        self._lsb_sd1_dout_delay1 = np.repeat(mpf(0.0), self._lsb_bits)
        self._lsb_sd1_dout_delay2 = np.repeat(mpf(0.0), self._lsb_bits)
        self._lsb_sd2_dout_delay1 = np.repeat(mpf(0.0), self._lsb_bits)
        self._lsb_sd2_dout_delay2 = np.repeat(mpf(0.0), self._lsb_bits)
        self._dwa_pointer = 0
        self._mes_selector = False
        self._chopper_state = 0
        self._build_capacitor_array()
        self._comparator_offset = high_precision_normal(0, 1e-3)
        self._loopfilter_offset = high_precision_normal(0, 1e-3)
        self._comparator_noise = 1e-4

    @property
    def switching_energy(self) -> float:
        """Switching energy dissipated per conversion."""
        return self._vdd ** 3.2 * self.total_capacitance

    @property
    def _time_constant(self) -> float:
        return 12.0

    @property
    def analog_energy(self) -> float:
        """Analog energy dissipated per conversion."""
        # 1/time_constant = 2 * currnent / 1.2 * 26e-3 / self.total_capacitance
        # based on empirical fitting 
        return self._vdd * 4 * 24 * self.total_capacitance * 1.2 * 0.1 / self._time_constant
    
    @property
    def digital_energy(self) -> float:
        """Digital energy dissipated per conversion."""
        # Based on empirical scaling and estimate for CIC
        gate_count = self.N * ( 3 + 5 ) * 25
        return gate_count * self._vdd ** 2 * 5e-15

    @property
    def comparator_output(self) -> bool:
        return (
            self.dac_output
            - high_precision_normal(self._comparator_offset, self._comparator_noise)
        ) >= 0
    
    @property
    def quantization_residue(self) -> float:
        return self._loopfilter_offset - self.dac_output

    @property
    def sar_residue(self) -> float:
        return self._loopfilter_offset - self.sar_output

    @classmethod
    def _limited_gain_integration(cls, state, new_val, gain) -> float:
        return (state+new_val) * gain / (1 + gain)

    def update_loopfilter(self) -> None:
        gain_factor_A1 = self.total_capacitance / self._qnf_caps[0] 
        gain_factor_A2 = 0.5 * self._qnf_caps[2] / self._qnf_caps[1]
        gain_factor_A3 = self._qnf_caps[4] / self._qnf_caps[3]
        self._qnf_aout[1] = self._limited_gain_integration(
            state=self._qnf_aout[1],
            new_val=self._qnf_aout[0] * gain_factor_A3 + self._sample_ktc_noise(self._qnf_caps[3]),
            gain=self._amplifier_gain
        )
        self._qnf_aout[0] = self._limited_gain_integration(
            state=self._qnf_aout[0],
            new_val=-self.sar_residue * gain_factor_A1 * gain_factor_A2 + self._sample_ktc_noise(self._qnf_caps[2]),
            gain=self._amplifier_gain
        )
        self._qnf_aout = self._qnf_aout.clip(-0.6, 0.6)

    def inc_pointer(self):
        self._dwa_pointer = (
            self._dwa_pointer 
            + self.msb_value
        ) % 2 ** self._msb_bits



    def _build_capacitor_array(self):

        # Calculate Array Weights
        lsb_weights = np.fromiter(
            ( mpf(2**elem) for elem in range(self._lsb_bits - 1, -1, -1)), 
            dtype=object)
        qnf_weights = np.fromiter(
            ( mpf(elem) for elem in (0.5,)*4 + (1,)*3), 
            dtype=object)

        # Sample Mismatch
        msb_mismatch = np.fromiter(
            (
                high_precision_normal(1, self._msb_mom_capacitor_mismatch_sigma_c18) 
                for _ in range(2 ** self._msb_bits - 1)
            ), 
            dtype=object
        )
        lsb_sd1_mismatch = np.fromiter(
            (
                high_precision_normal(
                    1, 
                    self._lsb_mom_capacitor_mismatch_sigma_c18 / weight ** 0.5
                ) 
                for weight in lsb_weights
            ), 
            dtype=object
        )
        lsb_sd2_mismatch = np.fromiter(
            (
                high_precision_normal(
                    1, 
                    self._lsb_mom_capacitor_mismatch_sigma_c18 / weight ** 0.5
                ) 
                for weight in lsb_weights
            ), 
            dtype=object
        )
        qnf_mismatch = np.fromiter(
            (
                high_precision_normal(
                    1, 
                    self._msb_mom_capacitor_mismatch_sigma_c18 / weight ** 0.5
                ) 
                for weight in qnf_weights
            ), 
            dtype=object
        )

        # Calculate Effective Weights
        self._msb_caps : npt.NDArray  = self.msb_unit_capacitor * msb_mismatch
        self._lsb_sd1_caps : npt.NDArray  = self.lsb_unit_capacitor * lsb_weights * lsb_sd1_mismatch
        self._lsb_sd2_caps : npt.NDArray  = self.lsb_unit_capacitor * lsb_weights * lsb_sd2_mismatch
        self._qnf_caps : npt.NDArray = self.msb_unit_capacitor * qnf_weights * qnf_mismatch

        self.dac_caps = np.concatenate(
            [
                self._msb_caps,
                self._lsb_sd1_caps,
                self._lsb_sd2_caps,
                self._qnf_caps[-2:],
            ]
        )

        self.dac_caps_ideal = np.concatenate(
            [
                np.repeat(self.msb_unit_capacitor, repeats=2 ** self._msb_bits - 1),
                self.lsb_unit_capacitor * lsb_weights,
                self.lsb_unit_capacitor * lsb_weights,
                self.msb_unit_capacitor * qnf_weights[-2:],
            ]
        )

    @property
    def qnf_capacitors(self) -> QnfCapacitors:
        return QnfCapacitors(
            *tuple(self._qnf_caps)
        )

        return np.sum(self.dac_caps)

    @property
    def _msb_unit_capacitor_area(self):
        """Capacitor area in um."""
        return self.msb_unit_capacitor / self._c_density

    @property
    def _msb_mom_capacitor_mismatch_sigma_c18(self):
        """Mismatch sigma from MC simulation. 0.1% for 10x10 m."""
        reference_area = 831.9e-15/self._c_density # 20.4 by 20.4 finger
        reference_sigma = (349.3e-3/831.9) # MC simulation
        return reference_sigma * np.sqrt(reference_area / self._msb_unit_capacitor_area)

    @property
    def _lsb_unit_capacitor_area(self):
        """Capacitor area in um."""
        return self.lsb_unit_capacitor / self._c_density

    @property
    def _lsb_mom_capacitor_mismatch_sigma_c18(self):
        """Mismatch sigma from MC simulation. 0.1% for 10x10 m."""
        reference_area = 51.42e-15/self._c_density # 20.4 by 20.4 finger
        reference_sigma = (88.17e-3/51.42) # MC simulation
        return reference_sigma * np.sqrt(reference_area / self._lsb_unit_capacitor_area)


    def _sample_ktc_noise(self, sampling_capacitance:float, noise_excess_factor: float = 1.0 ) -> float:
        sampling_sigma = np.sqrt( noise_excess_factor * self._kT / sampling_capacitance )
        return high_precision_normal(sigma=sampling_sigma) 

    def _sample(self, vin:float):
        """Stores the input voltage to be sampled by the DAC."""
        analog_bits = np.concatenate(
            [   # Pre Charge top cap to vdd/2 and sample control signals
                np.repeat(vin, self._msb_caps.size), 
                self._vdd/2 - self._lsb_sd1_dout*self._vdd, 
                self._vdd/2 - self._lsb_sd2_dout*self._vdd,
                self._qnf_aout*0,
            ]
        )
        noise = self._sample_ktc_noise(self.total_capacitance, noise_excess_factor = 3.0)
        self._v_sampled = (
            np.dot(analog_bits, self.dac_caps) / np.sum(self.dac_caps)
        ) + noise + self._loopfilter_offset - self._vdd/2
        return noise

    @property
    def _msb_encoded(self) -> npt.NDArray:
        """Return thermometer-coded vector for MSB."""
        code = np.full_like(self._msb_caps, fill_value=0.5)
        for index in range(0, self._msb_bits):
            big_endian = self._msb_bits - index 
            selection = range( 2 ** (big_endian-1) - 1, 2 ** big_endian - 1)
            if self._msb_dout[index] > 0.75:
                code[selection] = 1
            if self._msb_dout[index] < 0.25:
                code[selection] = 0

        assert sum(code == 1) == self.msb_value
        # print(code)
        return np.roll(
            code,
            self._dwa_pointer,
        )


    @property
    def sar_output(self) -> float:
        """Returns DAC output voltage given a digital code in terms of N bits."""
        bits = np.concatenate(

                self._msb_encoded,
                self._lsb_sd1_dout,
                self._lsb_sd2_dout,
                self._qnf_aout*0,

        return (
            np.dot(bits, self.dac_caps) / np.sum(self.dac_caps)
        ) * self._vdd + self._v_sampled

            return (

        else:
            return (
                self.lsb_sd1_value / lsb_scale_dout

                + self.msb_value

    def dout_mes1(self) -> float:
        lsb_scale_dout = ( 2 ** self._lsb_bits)
        if self._mes_selector: # converting on SD2
            return (
                self.lsb_sd2_value / lsb_scale_dout
                - 2 * self.lsb_sd1_value_delay1 / lsb_scale_dout
                + self.msb_value

        else: # converting on SD1

                + self.msb_value

                + self.lsb_sd1_value_delay2 / lsb_scale_dout
                + self.msb_value

    def sar_process(self) -> float:
        # Conversion register should be zero'd
        for cycle in range(self._msb_dout.size):
            self._msb_dout[cycle] = 0 if self.comparator_output else 1
        for cycle in range(self._lsb_bits):
            if self._mes_selector: # converting on SD2
                self._lsb_sd2_dout[cycle] = 0 if self.comparator_output else 1
            else: # converting on SD1
                self._lsb_sd1_dout[cycle] = 0 if self.comparator_output else 1
        return self.quantization_residue



    def _adc_convertion(self):
        quantization_error = self.sar_process()
        self.update_loopfilter()
        self.inc_pointer()
        return quantization_error

class PepperDataConverter_V2(PepperDataConverter_V1):
    
    """
    This is a new version of the Pepper ADC with only 1st order MES 
    """


    def __init__(self, 
            vdd: float = 1.8,
        ):
        self.N = 10
        self.msb_unit_capacitor = mpf(2*832.0e-15)
        self.lsb_unit_capacitor = mpf(0.5*52.0e-15)
        self._v_sampled = 0.0
        self._c_density = mpf(2e-15)
        self._vdd = mpf(vdd)
        self._lsb_bits = 6
        self._msb_bits = 4
        self._msb_dout = np.repeat(mpf(0.0), self._msb_bits)
        self._lsb_sd1_dout = np.repeat(mpf(0.0), self._lsb_bits)
        self._qnf_aout = np.repeat(mpf(0.0), 2)
        self._lsb_sd1_dout_delay1 = np.repeat(mpf(0.0), self._lsb_bits)
        self._dwa_pointer = 0
        self._chopper_state = 0
        self._build_capacitor_array()
        self._comparator_offset = high_precision_normal(0, 1e-3)
        self._loopfilter_offset = high_precision_normal(0, 1e-3)
        self._comparator_noise = 1e-4

    def _build_capacitor_array(self):

        # Calculate Array Weights
        lsb_weights = np.fromiter(
            ( mpf(2**elem) for elem in range(self._lsb_bits - 1, -1, -1)), 
            dtype=object)
        qnf_weights = np.fromiter(
            ( mpf(elem) for elem in (0.5,)*5 + (1,) + (0.5,)), 
            dtype=object)

        # Sample Mismatch
        msb_mismatch = np.fromiter(
            (
                high_precision_normal(1, self._msb_mom_capacitor_mismatch_sigma_c18) 
                for _ in range(2 ** self._msb_bits - 1)
            ), 
            dtype=object
        )
        lsb_sd1_mismatch = np.fromiter(
            (
                high_precision_normal(
                    1, 
                    self._lsb_mom_capacitor_mismatch_sigma_c18 / weight ** 0.5
                ) 
                for weight in lsb_weights
            ), 
            dtype=object
        )
        qnf_mismatch = np.fromiter(
            (
                high_precision_normal(
                    1, 
                    self._msb_mom_capacitor_mismatch_sigma_c18 / weight ** 0.5
                ) 
                for weight in qnf_weights
            ), 
            dtype=object
        )

        # Calculate Effective Weights
        self._msb_caps : npt.NDArray  = self.msb_unit_capacitor * msb_mismatch
        self._lsb_sd1_caps : npt.NDArray  = self.lsb_unit_capacitor * lsb_weights * lsb_sd1_mismatch
        self._qnf_caps : npt.NDArray = self.msb_unit_capacitor * qnf_weights * qnf_mismatch

        self.dac_caps = np.concatenate(
            [
                self._msb_caps,
                self._lsb_sd1_caps,
                self._qnf_caps[-2:],
            ]
        )
        self.dac_caps_ideal = np.concatenate(
            [
                np.repeat(self.msb_unit_capacitor, repeats=2 ** self._msb_bits - 1),
                self.lsb_unit_capacitor * lsb_weights,
                self.msb_unit_capacitor * qnf_weights[-2:],
            ]
        )

    def zero_bits(self):
        self._msb_dout.fill(0.5)
        self._lsb_sd1_dout.fill(0.5)



    def _sample(self, vin:float):
        """Stores the input voltage to be sampled by the DAC."""
        analog_bits = np.concatenate(
            [   # Pre Charge top cap to vdd/2 and sample control signals
                np.repeat(vin, self._msb_caps.size), 
                self._vdd/2 - self._lsb_sd1_dout*self._vdd, 
                self._qnf_aout*0,
            ]
        )
        noise = self._sample_ktc_noise(self.total_capacitance, noise_excess_factor = 3.0)
        self._v_sampled = (
            np.dot(analog_bits, self.dac_caps) / np.sum(self.dac_caps)
        ) + noise + self._loopfilter_offset - self._vdd/2
        return noise

    @property
    def sar_output(self) -> float:
        """Returns DAC output voltage given a digital code in terms of N bits."""
        bits = np.concatenate(
            [
                self._msb_encoded,
                self._lsb_sd1_dout,
                self._qnf_aout*0,
            ]
        )
        return (
            np.dot(bits, self.dac_caps) / np.sum(self.dac_caps)
        ) * self._vdd + self._v_sampled

    @property
    def dac_output(self) -> float:
        """Returns DAC output voltage given a digital code in terms of N bits."""
        bits = np.concatenate(
            [
                self._msb_encoded,
                self._lsb_sd1_dout,
                self._qnf_aout,
            ]
        )
        return (
            np.dot(bits, self.dac_caps) / np.sum(self.dac_caps)
        ) * self._vdd + self._v_sampled

    @property
    def dac_output_ideal(self) -> float:
        """Returns DAC output voltage given a digital code in terms of N bits."""
        bits = np.concatenate(
            [
                self._msb_encoded, 
                self._lsb_sd1_dout, 
                self._qnf_aout,
            ]
        )
        return (
            np.dot(bits, self.dac_caps_ideal) / np.sum(self.dac_caps_ideal)
        ) * self._vdd + self._v_sampled

    @property
    def dout_mes0(self) -> float:
        lsb_scale_dout = ( 2 ** self._lsb_bits)
        return (
            self.lsb_sd1_value / lsb_scale_dout
            + self.msb_value
        ) * self._vdd / 2 ** self._msb_bits

    @property
    def dout_mes1(self) -> float:
        lsb_scale_dout = ( 2 ** self._lsb_bits)
        return (
            self.lsb_sd1_value / lsb_scale_dout
            + self.lsb_sd1_value_delay1 / lsb_scale_dout
            + self.msb_value
        ) * self._vdd / 2 ** self._msb_bits

    def sar_process(self) -> float:
        # Conversion register should be zero'd
        for cycle in range(self._msb_dout.size):
            self._msb_dout[cycle] = 0 if self.comparator_output else 1
        for cycle in range(self._lsb_bits):
            self._lsb_sd1_dout[cycle] = 0 if self.comparator_output else 1
        return self.quantization_residue
